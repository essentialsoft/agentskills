# Skills Hub Web App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy a Vite + React static web app on Vercel that live-loads ESI skills from GitHub, supports search/filter, and provides detail pages with copyable install commands.

**Architecture:** A client-rendered React app with React Router uses a small data layer (GitHub API client + markdown parser + normalization helpers). The catalog route (`/`) renders searchable/filterable skill cards, and the detail route (`/skills/:name`) renders one skill, including deep-link fetch fallback. Error/empty/loading states are explicit and non-blocking.

**Tech Stack:** Vite, React, TypeScript, React Router, Vitest, React Testing Library, Vercel static hosting

```ts
import type { ParsedSkill } from "../types/skill";

function readFrontmatterValue(frontmatter: string, key: string): string {
  const pattern = new RegExp(`^${key}:\\s*(.+)$`, "m");
  const match = frontmatter.match(pattern);
  if (!match) return "";
  return match[1].trim().replace(/^"|"$/g, "");
}

export function parseSkillMarkdown(markdown: string): ParsedSkill | null {
  const fmMatch = markdown.match(/^---\n([\s\S]*?)\n---/);
  if (!fmMatch) return null;

  const frontmatter = fmMatch[1];
  const name = readFrontmatterValue(frontmatter, "name");
  if (!name) return null;

  const description = readFrontmatterValue(frontmatter, "description") || "No description provided.";
  const argumentHint = readFrontmatterValue(frontmatter, "argument-hint") || "";

  const purposeMatch = markdown.match(/# Purpose\n\n([\s\S]*?)(\n# |$)/);
  const purpose = purposeMatch ? purposeMatch[1].trim() : "Purpose section not available.";

  return {
    name,
    description,
    argumentHint,
    purpose
  };
}
```

```ts
import { parseSkillMarkdown } from "./parseSkillMarkdown";
import { toSkill } from "./skillTransforms";
import type { Skill } from "../types/skill";

const OWNER = "essentialsoft";
const REPO = "agentskills";
const BASE = `https://api.github.com/repos/${OWNER}/${REPO}/contents/skills`;

type GitHubEntry = {
  name: string;
  path: string;
  type: "file" | "dir";
};

async function fetchJson<T>(url: string): Promise<T> {
  const response = await fetch(url);
  if (!response.ok) {
    const message = response.status === 403 ? "rate_limit" : "request_failed";
    throw new Error(message);
  }
  return (await response.json()) as T;
}

async function fetchSkillMarkdown(path: string): Promise<string> {
  const fileUrl = `https://raw.githubusercontent.com/${OWNER}/${REPO}/main/${path}`;
  const response = await fetch(fileUrl);
  if (!response.ok) throw new Error("skill_markdown_failed");
  return await response.text();
}

export async function fetchAllSkills(): Promise<Skill[]> {
  const entries = await fetchJson<GitHubEntry[]>(BASE);
  const dirs = entries.filter((entry) => entry.type === "dir");

  const settled = await Promise.allSettled(
    dirs.map(async (dir) => {
      const path = `${dir.path}/SKILL.md`;
      const markdown = await fetchSkillMarkdown(path);
      const parsed = parseSkillMarkdown(markdown);
      if (!parsed) return null;
      return toSkill(parsed, path);
    })
  );

  return settled
    .filter((result): result is PromiseFulfilledResult<Skill | null> => result.status === "fulfilled")
    .map((result) => result.value)
    .filter((skill): skill is Skill => Boolean(skill));
}

export async function fetchSkillBySlug(slug: string): Promise<Skill | null> {
  const path = `skills/${slug}/SKILL.md`;
  const markdown = await fetchSkillMarkdown(path);
  const parsed = parseSkillMarkdown(markdown);
  return parsed ? toSkill(parsed, path) : null;
}
```

Create `apps/skills-hub/src/hooks/useSkills.ts`:

```ts
import { useCallback, useEffect, useMemo, useState } from "react";
import type { Skill } from "../types/skill";
import { fetchAllSkills } from "../lib/githubApi";

export type SkillsStatus = "loading" | "success" | "error";

export function useSkills() {
  const [skills, setSkills] = useState<Skill[]>([]);
  const [status, setStatus] = useState<SkillsStatus>("loading");
  const [errorCode, setErrorCode] = useState<string>("");

  const load = useCallback(async () => {
    setStatus("loading");
    setErrorCode("");
    try {
      const data = await fetchAllSkills();
      setSkills(data);
      setStatus("success");
    } catch (error) {
      const code = error instanceof Error ? error.message : "unknown_error";
      setErrorCode(code);
      setStatus("error");
    }
  }, []);

  useEffect(() => {
    void load();
  }, [load]);

  return useMemo(
    () => ({ skills, status, errorCode, retry: load }),
    [skills, status, errorCode, load]
  );
}
```

```ts
import type { Skill } from "../types/skill";

export function filterSkills(skills: Skill[], query: string, tag: string): Skill[] {
  const q = query.trim().toLowerCase();

  return skills.filter((skill) => {
    const matchesQuery =
      q.length === 0 ||
      skill.name.toLowerCase().includes(q) ||
      skill.description.toLowerCase().includes(q);

    const matchesTag = tag === "all" || skill.tags.includes(tag);

    return matchesQuery && matchesTag;
  });
}
```

```tsx
import { useMemo, useState } from "react";
import { useSkills } from "../hooks/useSkills";
import { filterSkills } from "../lib/filterSkills";
import { SearchFilterBar } from "../components/SearchFilterBar";
import { SkillCard } from "../components/SkillCard";
import { StatusState } from "../components/StatusState";

export function CatalogPage() {
  const { skills, status, errorCode, retry } = useSkills();
  const [query, setQuery] = useState("");
  const [tag, setTag] = useState("all");

  const tags = useMemo(() => {
    const all = new Set<string>();
    skills.forEach((skill) => skill.tags.forEach((item) => all.add(item)));
    return Array.from(all).sort();
  }, [skills]);

  const filtered = useMemo(() => filterSkills(skills, query, tag), [skills, query, tag]);

  if (status === "loading") {
    return <StatusState title="Loading skills" message="Fetching live skill catalog from GitHub..." />;
  }

  if (status === "error") {
    const message = errorCode === "rate_limit"
      ? "GitHub rate limit reached. Please retry in a few minutes."
      : "Could not load skills from GitHub.";
    return <StatusState title="Unable to load skills" message={message} onRetry={retry} />;
  }

  if (skills.length === 0) {
    return <StatusState title="No skills found" message="The repository did not return any skills." />;
  }

  return (
    <main>
      <h1>ESI Skills Hub</h1>
      <p>Browse available skills and open detail pages for usage guidance.</p>

      <SearchFilterBar query={query} onQueryChange={setQuery} tag={tag} onTagChange={setTag} tags={tags} />

      {filtered.length === 0 ? (
        <StatusState title="No matching skills" message="Try a different search term or clear filters." />
      ) : (
        <section className="skills-grid" aria-label="Skills catalog">
          {filtered.map((skill) => (
            <SkillCard key={skill.slug} skill={skill} />
          ))}
        </section>
      )}
    </main>
  );
}
```

```tsx
import { useState } from "react";

type CopyCommandProps = {
  command: string;
};

export function CopyCommand({ command }: CopyCommandProps) {
  const [status, setStatus] = useState<"idle" | "copied" | "failed">("idle");

  async function handleCopy() {
    try {
      await navigator.clipboard.writeText(command);
      setStatus("copied");
    } catch {
      setStatus("failed");
    }
  }

  return (
    <section className="copy-command">
      <code>{command}</code>
      <button onClick={() => void handleCopy()}>Copy install command</button>
      {status === "copied" ? <p>Copied</p> : null}
      {status === "failed" ? <p>Copy failed. Please copy manually.</p> : null}
    </section>
  );
}
```

```tsx
import { Link, useParams } from "react-router-dom";
import { useEffect, useState } from "react";
import type { Skill } from "../types/skill";
import { fetchSkillBySlug } from "../lib/githubApi";
import { buildInstallCommand } from "../lib/installCommand";
import { CopyCommand } from "../components/CopyCommand";
import { StatusState } from "../components/StatusState";

export function SkillDetailPage() {
  const { slug = "" } = useParams();
  const [skill, setSkill] = useState<Skill | null>(null);
  const [status, setStatus] = useState<"loading" | "success" | "error" | "missing">("loading");

  useEffect(() => {
    async function load() {
      try {
        const found = await fetchSkillBySlug(slug);
        if (!found) {
          setStatus("missing");
          return;
        }
        setSkill(found);
        setStatus("success");
      } catch {
        setStatus("error");
      }
    }
    void load();
  }, [slug]);

  if (status === "loading") return <StatusState title="Loading skill" message="Fetching skill details..." />;
  if (status === "error") return <StatusState title="Unable to load skill" message="Please try again." />;
  if (status === "missing") return <StatusState title="Skill not found" message="The requested skill does not exist." />;
  if (!skill) return null;

  return (
    <main>
      <Link to="/">Back to catalog</Link>
      <h1>{skill.name}</h1>
      <p>{skill.description}</p>
      <h2>Purpose</h2>
      <p>{skill.purpose}</p>
      <h2>Argument hint</h2>
      <p>{skill.argumentHint || "No argument hint provided."}</p>
      <CopyCommand command={buildInstallCommand(skill.name)} />
    </main>
  );
}
```
